 18def heatmap(title, data, save, path=None, filename='map', dpi=120, **kwargs):
 19    """Generic heatmap function
 20
 21    Args:
 22        title (string): Title of the heatmap
 23        data (pd.DataFrame): The df to plot
 24        save (bool): Whether to save the heatmap
 25        dpi (int): Dots per inch to save the plot
 26        filename (str): The filename to save the plot
 27        path (str): Path to save the plots
 28        kwargs (dict): Additional keyword arguments for sns.heatmap
 29
 30    Returns:
 31        plt.figure: The figure handle
 32
 33    """
 34    rows, cols = len(data.index), len(data.columns)
 35    fig = plt.figure(figsize=(rows, cols / 4 + 2), dpi=dpi)
 36    sns.heatmap(data, annot=True, **kwargs)
 37    plt.title(title, fontweight='bold')
 38
 39    if path is None:
 40        spath = os.path.join(RESULTS_PATH, 'heatmaps', filename)
 41    else:
 42        spath = os.path.join(path, filename)
 43
 44    save_figure(save, dpi, spath, fig)
 45
 46    return fig

 49def correlation_heatmap(features=None, df=None, save=False, dpi=150,
 50                        filename='cor_heatmap', cbar=False, path=None,
 51                        all_with_these_features=False, method='pearson',
 52                        plot_corr=True, pvalues=True, significant=True,
 53                        plot_significant=True, threshold=10, min_cor=0):
 54    """Heatmap of all correlations of the prepared dataframe
 55
 56    Args:
 57        features (list): The features for which all correlations are shown
 58        df (any): The data to plot, if None it will be loaded from data prep
 59        save (bool): Whether to save the plot
 60        dpi (int): Dots per inch to save the plot
 61        filename (str): The filename to save the plot
 62        cbar (bool): Whether to show the colorbar
 63        path (str): Path to save the plots
 64        all_with_these_features (bool): Show all cors of given features
 65        method (str): Either 'pearson' or 'spearman' for the correlation type
 66        plot_corr (bool): Whether to plot the correlation heatmap
 67        pvalues (bool): Whether to show the p-values as separate heatmap
 68        significant (bool): Whether to show the significant correlation values
 69        plot_significant (bool): Whether to plot the significant correlations'
 70        threshold (float): The threshold value for the correlations
 71        min_cor (float): The minimum correlation
 72
 73    """
 74    if df is None:
 75        df = load_data('data_prep_with_personal.csv')
 76
 77    df = df.drop(['User'], axis=1)
 78
 79    if features is None:
 80        features = df.columns
 81
 82    if isinstance(features, str):
 83        features = [features]
 84
 85    if not all_with_these_features:
 86        df = df[features]
 87
 88    pearson = df.corr(method='pearson')[features]
 89    spearman = df.corr(method='spearman')[features]
 90    correlation = pearson if method == 'pearson' else spearman
 91    figures = []
 92
 93    if plot_corr:
 94        kwargs = dict(fmt='.3f', vmin=-1, vmax=1, cbar=cbar, filename=filename)
 95        fig = heatmap('Feature correlations', correlation, save, **kwargs)
 96        figures.append(fig)
 97
 98    if pvalues or significant:
 99
100        p_pearson = 100 * df.corr(method=cor_test('pearson'))[features]
101        p_spearman = 100 * df.corr(method=cor_test('spearman'))[features]
102        p = p_pearson if method == 'pearson' else p_spearman
103
104        cmap = sns.diverging_palette(133, 10, as_cmap=True)
105
106        if pvalues:
107            kwargs = dict(fmt='.2f', vmin=0, vmax=100, cbar=cbar, cmap=cmap,
108                          filename=filename+'_pvalues')
109            title = 'Two sided feature correlation p-values / %'
110            fig = heatmap(title, p, save, **kwargs)
111            figures.append(fig)
112
113        if significant:
114
115            args = (method, pearson, p_pearson, spearman, p_spearman,
116                    threshold, min_cor)
117            df, numeric = get_significant_correlations(*args)
118
119            numeric = numeric.drop(['Prio 1', 'Prio 2'], axis=1)
120            df = df.drop(['Prio 1', 'Prio 2'], axis=1)
121
122            if len(df.index) > 0 and plot_significant:
123                fig = plot_significant_corrs(numeric, df, dpi, save, path,
124                                             filename + '_sgnf')
125                figures.append(fig)
126
127    plt.show()
128    return figures, df

184def cluster_pairs_plot(pairs: list[pd.DataFrame], labels: pd.DataFrame,

186                       user_ids=None, show_cluster_of=None) -> list:
187    """Plot the kmeans clustering result for a pair of features

190        pairs (list[pd.DataFrame]): The input dataframes to be plotted

245def cluster_single_plot(columns: list[pd.DataFrame], labels: pd.DataFrame,
246                        save=True, path="", dpi=300, n_bins=20) -> list:
247    """Plot the kmeans clustering result for single features
248
249    Args:
250        columns (list[pd.DataFrame]): The input dataframes to be plotted
251        labels (list): List of pandas series with labels for each kmeans result
252        save (bool): Whether to save the figure
253        path (string): Path to save the figure
254        dpi (int): Dots per inch for saving the figure
255        n_bins (int): Number of bins to show in histogram
256        
257    Returns:
258        list: List of axes of the plots
259
260    """
261    ax = init_subplots_plot(len(columns))
262
263    for idx, data in enumerate(columns):
264        c_ax = get_current_axis(list(labels.columns), ax, idx)
265        data = data.reset_index().drop('index', axis=1)
266        tag = data.columns[0]

268        data = pd.concat([data.iloc[:, 0], pd.Series(label)], axis=1)
269        data.columns = [tag, 'label']

278                         multiple='stack', bins=n_bins, palette="tab10",
279                         legend=legend)
280        c_ax.set_title(f"{tag}", fontweight='semibold')

285    save_figure(save, dpi, os.path.join(path, "histogram"))
286    plt.show()

437def significant_corrs_heatmap(method='pearson', p_threshold=5, min_cor=0,
438                              cor_threshold=0.8, save=True, path=''):
439    """Plot an overview heatmap for significant correlations
440
441    Args:
442        method (str): The type of correlation to plot (spearman, pearson)
443        p_threshold (float): The threshold for the p-value in %
444        min_cor (float): The minimum correlation to consider
445        cor_threshold (float): The threshold for abs(correlation)
446        save (bool): Whether to save the plot
447        path (str): Path to save the plots
448
449    Returns:
450        plt.figure: The figure handle
451
452    """
453    prios = load_data('features_categories.csv', rm_first_col=False)
454    features = list(prios.loc[prios.Prio != 0, 'Feature'])
455    path = os.path.join(path, "Heatmaps")
456
457    kwargs = dict(plot_corr=False, pvalues=False, significant=True,
458                  method=method, plot_significant=False, threshold=p_threshold,
459                  min_cor=min_cor)
460
461    _, df = correlation_heatmap(features=features, save=False, **kwargs)
462
463    for idx in df.index:
464
465        prio1, prio2 = df.loc[idx, "Prio 1"], df.loc[idx, "Prio 2"]
466        feature1, feature2 = df.loc[idx, "Feature 1"], df.loc[idx, "Feature 2"]
467        corr = df.loc[idx, method]
468
469        if abs(corr) > cor_threshold:
470            if prio2 >= prio1:
471                if feature1 in features:
472                    features.remove(feature1)
473            elif prio1 > prio2:
474                if feature2 in features:
475                    features.remove(feature2)
476
477    kwargs['plot_significant'] = True
478
479    fig, _ = correlation_heatmap(features=features, save=save, path=path,
480                                 **kwargs)
481
482    return fig, features

485def get_success_moderators(diff='RF_Diff', splits=2, save=False, plot=True,
486                           sort_col='meanchange', drop_fits=True):
487    df = load_data('data_prep_with_personal.csv')
488    obsolete = 'NSt_Diff' if diff == 'RF_Diff' else 'RF_Diff'
489    df = df.drop(['User', obsolete], axis=1)
490
491    diff_mean, diff_std = df[diff].mean(), df[diff].std()
492
493    res = {}
494
495    for col in list(df.columns):
496
497        if col == diff:
498            continue
499
500        current_df = df.dropna(subset=[col, diff])
501        current = current_df[col]
502        n = len(current)
503
504        lq = list(current.sort_values())[n // splits]
505        uq = list(current.sort_values())[n - n // splits]
506
507        lb = current_df.loc[current <= lq, diff].mean()
508        ub = current_df.loc[current >= uq, diff].mean()
509        mean_diff = (ub - lb) / diff_std * np.sqrt(n // splits) / 2
510
511        spear = stats.spearmanr(current, current_df[diff])
512        pears = stats.pearsonr(current, current_df[diff])
513
514        res[col] = (mean_diff, *spear, *pears)
515
516    cols = ("meanchange", "spear", "spear_p", "pears", "pears_p")
517    moderators = pd.DataFrame(res.values(), index=res.keys(), columns=cols)
518    moderators = moderators.sort_values(by=sort_col, ascending=False)
519
520    if drop_fits:
521        moderators = moderators.drop([x
522                                      for x in moderators.index if 'Fit' in x])
523
524    prios = load_data('features_categories.csv',
525                      rm_first_col=False).set_index('Feature')
526
527    numeric = moderators.copy().reset_index()
528    feature_prios = []
529    for feature in numeric['index']:
530        if feature in prios.index:
531            feature_prios.append(prios.loc[feature].values[0])
532        else:
533            feature_prios.append(3)
534    numeric['index'] = feature_prios
535
536    numeric = numeric.rename(columns={'index': 'Feature'})
537    df = moderators.copy().reset_index().rename(columns={'index': 'Feature'})
538
539    if plot:
540        plot_significant_corrs(numeric, df, save=save,
541                               filename=diff + '_moderators',
542                               title='Influence of Features on ' + diff)
543
544    return moderators

547def simulate_random_influences(reps) -> pd.DataFrame:
548    """Simulate random influences (corr etc.) of features on learn success
549
550    Args:
551        reps (int): Number of random features to simulate
552
553    Returns:
554        pd.DataFrame: Structure as the one returned by get_success_moderators
555
556    """
557    diff = load_data('data_prep_with_personal.csv')['RF_Diff']
558    idx = diff.index
559    std = diff.std()
560    n = len(idx)
561
562    random_moderators = {'meanchange': [], 'pears': [], 'spear': []}
563
564    for _ in range(reps):
565        random.shuffle(diff)
566        mean_diff = (np.mean(diff[:n // 2]) - np.mean(diff[n // 2:]))
567        mean_diff_scaled = mean_diff / std * np.sqrt(n // 2) / 2
568        random_moderators['meanchange'].append(mean_diff_scaled)
569        random_moderators['pears'].append(stats.pearsonr(diff, idx)[0])
570        random_moderators['spear'].append(stats.spearmanr(diff, idx)[0])
571
572    return pd.DataFrame(random_moderators)

575def summarize_feature_influences(moderators, diff='RF_Diff',
576                                 save=False) -> Tuple[plt.Figure, np.ndarray]:
577    """Summarize cor, ... with learn success for features in Histogram + qqPlot
578
579    Args:

581        diff (string): The type of diff measurement to consider

583
584    Returns:
585        Tuple[plt.Figure, np.ndarray]: The figure and axis handle
586
587    """
588    random_mods = simulate_random_influences(len(moderators.index))
589
590    labs = ["success change high/low feature values / std",
591            "Pearson correlation feature vs learn success",
592            "Spearman correlation feature vs learn success"]
593
594    fig, ax = plt.subplots(3, 3, figsize=(10, 9), dpi=200)
595
596    for idx, col in enumerate(['meanchange', 'pears', 'spear']):

600
601        stats.probplot(moderators[col] / np.std(moderators[col]),
