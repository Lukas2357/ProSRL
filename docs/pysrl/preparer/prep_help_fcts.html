 15def get_duplicates(df: pd.DataFrame) -> pd.Series:
 16    """Get number of duplicated labels for each user
 17
 18    Args:
 19        df (pd.DataFrame): The transformed dataframe
 20
 21    Returns:
 22        pd.Series: The number of duplicated labels for each user
 23
 24    """
 25    duplicates = df.groupby(['User', 'Label']).size().reset_index()
 26
 27    duplicates = duplicates[duplicates[0] > 1].drop('Label', axis=1)
 28    duplicates[0] = duplicates[0] - 1
 29
 30    duplicates = duplicates.groupby('User').sum()
 31    duplicates = duplicates.reindex(list(range(df['User'].nunique())),
 32                                    fill_value=0)
 33
 34    return duplicates

 54def get_learn_runs(learn_types: pd.Series, times: pd.Series) -> pd.DataFrame:
 55    """Get a list with length of all runs for the given learn types of a user
 56
 57    Args:
 58        learn_types (pd.Series): Series of learn activities of a user
 59        times (pd.Series): Series of activity times of a user
 60
 61    Returns:
 62        pd.DataFrame: Learn runs for the user
 63
 64    """
 65    learn_types, times = list(learn_types), list(times)
 66
 67    runs = []
 68    current_learn_type = learn_types[0]
 69    run_length = 1
 70    run_time = times[0]
 71
 72    for idx, i in enumerate(learn_types[1:]):
 73        if i == current_learn_type:
 74            run_length += 1
 75            run_time += times[idx + 1]
 76        else:
 77            runs.append((run_length, run_time))
 78            current_learn_type = i
 79            run_length = 1
 80            run_time = times[idx + 1]
 81    runs.append((run_length, run_time))
 82
 83    return pd.DataFrame(runs, columns=['acts', 'time'])

 86def predict_chance(arr: any, known=1) -> float:
 87    """Calculate the chance to predict the next value in an array
 88
 89    Args:
 90        arr (any): Any iterable object
 91        known (int): The number of values known to predict the next
 92
 93    Returns:
 94        float: The chance that we predict the right next value if we are given 
 95                a random value in the array
 96    
 97    """
 98    count = {}
 99    for idx, i in enumerate(arr[:-known]):
100        given = tuple(arr[k] for k in range(idx, idx+known))
101        if given in count.keys():
102            count[given].append(arr[idx+known])
103        else:
104            count[given] = [arr[idx+known]]
105    result = 0
106    for _, value in count.items():
107        result += max(dict(Counter(value)).values())
108        
109    return result/(len(arr)-known)

112def _entropy(p_i: list) -> float:
113    """Get the entropy for probabilities p_i
114    
115    Args:
116        p_i (list): List of probabilities for outcomes of a random variable
117
118    Returns:
119        float: The entropy of the distribution
120    
121    """
122    if len(p_i) == 0:
123        return 0
124    if isinstance(p_i[0], float):
125        n_variables = len(p_i)
126    else:
127        n_variables = len(p_i[0])
128        
129    p_i = np.array(p_i)
130    p_i = p_i[p_i != 0]
131    p_i = p_i[p_i != 1]
132    
133    if len(p_i) < 2:
134        return 0
135    
136    return np.sum(-p_i*np.log(p_i))/np.log(n_variables)

139def cond_entropy(arr: np.array, distance=0) -> float:
140    """Get the conditional entropy for an array and a given distance
141    
142    Args:
143        arr (np.array): An array to get conditional entropies for
144        distance (int): The number of points known 
145
146    Returns:
147        float: The conditional entropy of the array
148    
149    """
150    unique, counts = np.unique(arr, return_counts=True)
151    
152    if distance == 0:
153        return _entropy(counts / len(arr))
154    
155    cond_entr = 0
156    
157    for y in product(*repeat(unique, distance)):
158        
159        arr = list(arr)
160        
161        cond_ps = []
162        y_str = ', '.join([str(i) for i in y])
163        y_occ = len(re.findall(fr'{y_str}', repr(arr), overlapped=True))
164        
165        for x in unique:
166            yx_occ = len(re.findall(fr'{y_str}, {x}', repr(arr), 
167                                    overlapped=True))
168            if y_occ:
169                cond_ps.append(yx_occ/y_occ)
170                
171        tot_prob = sum(cond_ps)
172        
173        if tot_prob:
174            cond_ps = [c/tot_prob for c in cond_ps]
175            
176        cond_entr += y_occ * _entropy(cond_ps) / len(arr)
177        
178    return cond_entr

261def freq_features(user_dfs: list, use_time=False, used_learntypes=None,
262                  merge=False) -> pd.DataFrame:
263    """Get the features about the frequency of learn task changes
264
265    Args:
266        user_dfs (list): List of df for each user
267        use_time (bool): Whether to use the time spent as weights
268        used_learntypes (any): Learntypes included in the analysis
269        merge (bool) Whether to merge K+B+BK_Info and Ü+Ü_Info
270
271    Returns:
272        pd.DataFrame: The df of frequency features for each user
273
274    """
275    pd.options.mode.chained_assignment = None
276    largest_runs, mean_run_length = [], []
277    used = [0, 2, 3, 5] if used_learntypes is None else used_learntypes
278
279    for user_df in user_dfs:
280        user_df = user_df[user_df.LearnType.isin(used)]
281        if merge:
282            mapping = dict(zip(range(7), [0, 1, 1, 2, 2, 2, 2]))
283            user_df['LearnType'] = user_df['LearnType'].map(mapping)
284
285        runs = get_learn_runs(user_df["LearnType"], user_df['SecSpent'])
286        runs['count'] = runs['acts']
287
288        grouped_runs = runs[['count', 'acts', 'time']].groupby('acts').sum()
289        col = 'time' if use_time else 'count'
290
291        largest_two = grouped_runs[col].sort_values(ascending=False).iloc[:2]
292        length_norm = [item / sum(grouped_runs[col]) for item in largest_two]
293        largest_runs.append(tuple(zip(largest_two.index, length_norm)))
294        mean_run_length.append(runs[col].mean())
295
296    run_data = zip(mean_run_length, largest_runs)
297    features = []
298    for mean, f in run_data:
299        if len(f) == 1:
300            f = f + f
301        features.append((mean, f[0][0], f[1][0],
302                         (f[0][0] + f[1][0]) / 2,
303                         f[0][1] * 100, f[1][1] * 100,
304                         f[0][1] * 100 + f[1][1] * 100))
305
306    labels = ["fMean", "f1", "f2", "f12", "f1Amp", "f2Amp", "f12Amp"]
307    return pd.DataFrame(features, columns=labels)

310def time_dependent_features(user_dfs: list, labels: list, f_typ='LearnType',
311                            mapping=None, split_cats=False, plot=False,
312                            n_min=3, dpi=200, save=False, n_ranges=2,
313                            path='', threshold=2) -> pd.DataFrame | list:
314    """Get time dependent features for each user by temporal regression
315
316    Args:
317        user_dfs (list): The dfs with activities of each user
318        labels (list): The labels for the new features generated
319        f_typ (str, optional): The feature type used. Defaults to 'learntype'.
320        mapping (list, optional): The learntype mapping used. Defaults to None.
321        split_cats (bool, optional): Whether to split categories and get the 
322                                     mean. Defaults to False.
323        plot (bool, optional): Whether to generate plots. Defaults to False.
324        n_min (int, optional): Minimum points used in regression. Defaults to 3.
325        dpi (int): Dots per inch for saving the figure
326        save (bool): Whether to save the figure
327        n_ranges (int): The number of ranges to split time axis in
328        path (string): Path to save the figure
329        threshold (int): The threshold for the number of points needed for fit
330
331    Returns:
332        pd.DataFrame | list: The df with time dependent features of each user
333                             if plot is false, else the list of generated plots
334        
335    """
336    pd.options.mode.chained_assignment = None  # default='warn'
337
338    n_cats = 6 if split_cats else 1
339    fit_param = {f'{label} Fit': [] for label in labels}
340    if len(labels) > 1 and mapping is None:
341        mapping = list(range(len(labels)))
342
343    figures, n_artists = [], 1
344    fig, ax = None, None
345
346    for user_df in user_dfs:
347        
348        user_df = user_df[user_df.SecSpent < 1800]
349        user_df['UserCumSec'] = user_df['SecSpent'].cumsum()
350        user_df = user_df[user_df[f_typ] >= 0]
351
352        if mapping is not None and len(labels) > 1:
353            user_df = user_df[user_df[f_typ] < len(mapping)]
354            mapper = dict(zip(range(len(mapping)), mapping))
355            user_df[f_typ] = user_df[f_typ].map(mapper)
356        else:
357            mapping = range(len(labels))
358
359        if plot and n_artists > 0:
360            fig, ax = plt.subplots(1, 1, figsize=(10, 3.5), dpi=dpi)
361
362        y_max, n_artists = 0, 0
363        if len(labels) == 1:
364            x = np.array(user_df.UserCumSec) / 60
365            y = np.array(user_df[f_typ])
366            if len(y) > 0:
367                y_max = max(y_max, max(y))
368            x_ranges, y_ranges = get_ranges(x, n_ranges, y)
369            if len(x) > 0:
370                a, w = add_regression(x, x, y, y, ax, [], [], plot, n_min,
371                                      x_ranges, y_ranges,
372                                      np.array([0]*(n_ranges+1)), 'k',
373                                      labels[0])
374                n_artists += 1
375            else:
376                a = [0]
377            fit_param[f'{labels[0]} Fit'].append(a[0]*100)
378
379        else:
380            values = dict(Counter(mapping)).keys()
381            prev_range_data = np.zeros(n_ranges+1)
382
383            for idx, (label, value) in enumerate(zip(labels, values)):
384
385                a, w = [], []
386                color = list(colors.TABLEAU_COLORS.values())[idx]
387
388                for cat in range(n_cats):
389
390                    data = user_df[user_df.Category == cat] if split_cats \
391                        else user_df
392                    data['UserCumFeatureSec'] = data['SecSpent'].cumsum()
393
394                    if len(data.index) > threshold*len(labels):
395
396                        data[label] = user_df['SecSpent']
397                        data.loc[user_df[f_typ] != value, label] = 0
398                        label_actions = data.index[data[label] > 0]
399                        data[label] = data[label].cumsum()
400
401                        x = np.array(data.UserCumSec) / 60
402                        y = np.array(data[label])
403
404                        norm = np.array(data['UserCumFeatureSec'])
405                        x_ranges, y_ranges = get_ranges(x, n_ranges, y, norm)
406
407                        y_line, y_marker = y / norm, data[label] / norm
408                        x_marker = data.UserCumSec.loc[label_actions] / 60
409                        y_marker = np.array(y_marker.loc[label_actions])
410
411                        if len(y_line) > 0:
412                            y_max = max(y_max, max(y_line))
413
414                        if len(x) > 0:
415                            a, w = add_regression(x, x_marker, y_line,
416                                                  y_marker, ax, a, w,
417                                                  plot, n_min, x_ranges,
418                                                  y_ranges, prev_range_data,
419                                                  color, labels[idx])
420                            n_artists += 1
421                        else:
422                            a.append(0)
423                            w.append(0)
424                        prev_range_data += y_ranges
425
426                a_mean = 100*np.average(a, weights=w) if a else 0
427
428                fit_param[f'{label} Fit'].append(a_mean)
429            
430        if plot:
431            plt.xlabel('Zeit des Users auf betrachteten Seiten / min')
432            if len(labels) > 1:
433                plt.ylabel('Anteil an allen bisherigen Aktionen')
434            if len(user_df['UserCumSec']) > 0:
435                if max(user_df['UserCumSec']) > 0:
436                    plt.xlim([0, max(user_df['UserCumSec'])/60])
437                if y_max > 0:
438                    plt.ylim([-0.02, y_max*1.02])
439            if n_artists > 0:
440                plt.legend(ncol=len(labels))
441                user_id = str(min(user_df.User))
442                save_figure(save=save, path=os.path.join(path, user_id),
443                            dpi=dpi, fig=fig, recent=False)
444                figures.append((fig, ax))
445
446    if plot:
447        return figures
448            
449    return pd.DataFrame(fit_param)

452def get_ranges(x: ArrayLike, n_ranges: int, y: ArrayLike,
453               norm: ArrayLike = None) -> tuple[np.array, np.array]:
454    """Split x and y points into n ranges of same x size
455
456    Args:
457        x (ArrayLike): The x values to use
458        n_ranges (int): The number of ranges to use
459        y (ArrayLike): The y values to use
460        norm (ArrayLike): The normalization values to use
461
462    """
463    x_ranges, y_ranges, norm_ranges, k, prev_i = [], [], [], 0, 0
464    for i, val in enumerate(x):
465        if val >= k * max(x) / n_ranges:
466            x_ranges.append(k * max(x) / n_ranges)
467            y_correction = 0 if i == 0 else y_ranges[-1]
468            if norm is not None:
469                norm_correction = 0 if i == 0 else norm_ranges[-1]
470                norm_ranges.append(norm[i] - norm_correction)
471                y_ranges.append(y[i] - y_correction)
472            else:
473                if i == 0:
474                    y_ranges.append(0)
475                else:
476                    y_ranges.append(np.mean(y[prev_i:i+1]))
477            prev_i = i
478            k += 1
479        if k > n_ranges:
480            break
481    l_diff = n_ranges + 1 - len(x_ranges)
482    if l_diff > 0:
483        if len(x_ranges) > 0:
484            x_ranges.extend([x[-1]]*l_diff)
485            y_ranges.extend([y[-1]]*l_diff)
486            if norm is not None:
487                norm_ranges.extend([norm[-1]]*l_diff)
488
489    if norm is None:
490        return np.array(x_ranges), np.array(y_ranges)
491    else:
492        return np.array(x_ranges), np.array(y_ranges)/np.array(norm_ranges)

495def add_regression(x_line: np.array, x_marker: np.array, y_line: np.array,
496                   y_marker: np.array, axes: plt.Axes, a: list, weights: list,
497                   plot: bool, n_min: int, x_ranges: np.array,
498                   y_ranges: np.array, prev_range_data: np.array, color: any,
499                   label: str) -> tuple:
500    """Add a linear regression result and plot to a given list of results
501
502    Args:
503        x_line (np.array): x data to plot as line
504        x_marker (np.array): x data to plot with markers
505        y_line (np.array): y data to plot as line
506        y_marker (np.array): y data to plot with markers
507        axes (plt.Axes): The axis handle
508        a (list): List of slope parameters from linear regressions
509        weights (list): List of weights for each linear regression result
510        plot (bool): Whether to show the regression plot
511        n_min (int): Minimum number of data points required for regression
512        prev_range_data (np.array): Previous mean data of different ranges
513        x_ranges (np.array): Borders of different ranges
514        y_ranges (np.array): Mean data of different ranges
515        color (any): The color to use for plotting
516        label (str): Label for the legend
517
518    Returns:
519        tuple: The new axis handle and the extended result lists
520        
521    """
522
523    x_line = np.concatenate([[0], x_line])
524    y_line = np.concatenate([[y_line[0]], y_line])
525
526    reg = np.polyfit(x_line, y_line, 1) if len(x_line) >= n_min else [0, 0]
527
528    if plot:
529        axes.plot(x_line, y_line, ':', color=color, ms=8,
530                  label=label + ' Anteil')
531        axes.plot(x_marker, y_marker, '.', color=color, ms=8,
532                  label=label + ' Actions')
533        # axes.plot(x, np.polyval(reg, x), '--', color=color, label='_')
534        for i in range(1, len(y_ranges)):
535            y_min = prev_range_data[i]
536            if color == 'k':
537                y_min = y_min / max(y_line) if max(y_line) > 0 else y_min
538            y_max = (y_ranges[i]+prev_range_data[i])
539            if color == 'k':
540                y_max = y_max / max(y_line) if max(y_line) > 0 else y_max
541            axes.axvspan(
542                x_ranges[i-1], x_ranges[i], y_min, y_max,
543                alpha=0.35, color=color, label='_'
544            )
545
546    a.append(reg[-2])
547    weights.append(len(x_line))
548        
549    return a, weights

552def add_divided_columns(df, cols, div_col, perc=False, postfix='PerX'):
553    """Add columns to df generated by division by another column
554
555    Args:
556        df (pd.DataFrame): The df to use
557        cols (Iterable[str]): The columns to divide
558        div_col (Iterable(str) | str): The column(s) to divide by.
559            If str, all divided by that, else expect len(div_col) == len(cols)
560        perc (bool): Whether to make percentage (* by 100), defaults to False
561        postfix (str): The postfix to add to new column names, defaults to PerX
562
563    Returns:
564        pd.DataFrame: The concatenated df with new columns
565
566    """
567    if isinstance(div_col, str):
568        df_div_col = df[cols].div(df[div_col], axis=0)
569    else:
570        df_cols = df[cols]
571        df_cols.columns = div_col
572        df_div_col = df_cols.div(df[div_col])
573    df_div_col = df_div_col * 100 if perc else df_div_col
574    div_col = div_col if isinstance(div_col, str) else div_col[0]
575
576    if postfix == 'PerX':
577        if perc:
578            postfix = 'Perc'
579        elif 'Acts' in div_col:
580            postfix = '_TimePerAct'
581    elif postfix == '':
582        df = df.drop(cols, axis=1)
583
584    df_div_col.columns = [c + postfix for c in cols]
585    df = pd.concat([df, df_div_col], axis=1)
586
587    return df

590def get_fool_features(user_dfs: list, plot=False) -> pd.DataFrame:
591    """Get features indicating if user stupidly follows learning environment
592
593    Args:
594        user_dfs (list): The dfs with activities of each user
595        plot (bool): Whether to plot the features
596
597    """
598    pages = load_learn_pages("learn_pages_sorted")
599    features = pd.DataFrame()
600
601    for user_df in user_dfs:
602
603        line = []
604        for label in user_df.Label:
605            if label in pages.keys() and pages[label]["Title"]:
606                line.append(list(pages.keys()).index(label))
607
608        if plot:

610            plt.show()
611            plt.close()

487            if norm is not None:
488                norm_ranges.extend([norm[-1]]*l_diff)
489
490    if norm is None:
491        return np.array(x_ranges), np.array(y_ranges)
492    else:
